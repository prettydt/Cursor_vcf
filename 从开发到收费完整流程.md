#### 步骤7.2：添加授权验证逻辑（10分钟）
**操作**：

在 `<script>` 标签中添加以下代码：

```javascript
// 授权管理
const LICENSE_STORAGE_KEY = 'vcf_tool_license';
const USAGE_STORAGE_KEY = 'vcf_tool_usage';

// 检查是否是专业版用户
function isPremiumUser() {
    const license = localStorage.getItem(LICENSE_STORAGE_KEY);
    return license !== null && license.length > 0;
}

// 检查今日使用次数
function checkDailyUsage() {
    if (isPremiumUser()) {
        return { allowed: true, used: 0, limit: Infinity };
    }
    
    const today = new Date().toDateString();
    const usage = JSON.parse(localStorage.getItem(USAGE_STORAGE_KEY) || '{}');
    
    if (usage.date !== today) {
        usage.date = today;
        usage.count = 0;
    }
    
    const limit = 3;
    const used = usage.count || 0;
    
    return {
        allowed: used < limit,
        used: used,
        limit: limit
    };
}

// 增加使用次数
function incrementUsage() {
    if (isPremiumUser()) return;
    
    const today = new Date().toDateString();
    const usage = JSON.parse(localStorage.getItem(USAGE_STORAGE_KEY) || '{}');
    
    if (usage.date !== today) {
        usage.date = today;
        usage.count = 1;
    } else {
        usage.count = (usage.count || 0) + 1;
    }
    
    localStorage.setItem(USAGE_STORAGE_KEY, JSON.stringify(usage));
    updateUsageDisplay();
}

// 更新使用次数显示
function updateUsageDisplay() {
    const usageInfo = document.getElementById('usage-info');
    if (!usageInfo) return;
    
    if (isPremiumUser()) {
        usageInfo.textContent = '专业版：无限使用 ⭐';
        usageInfo.style.color = '#28a745';
    } else {
        const usage = checkDailyUsage();
        usageInfo.textContent = `今日已使用: ${usage.used}/${usage.limit}次`;
        usageInfo.style.color = usage.used >= usage.limit ? '#dc3545' : '#667eea';
    }
}

// 激活授权码
function activateLicense() {
    const input = document.getElementById('license-key');
    const status = document.getElementById('license-status');
    const license = input.value.trim();
    
    if (!license) {
        status.textContent = '请输入授权码';
        status.className = 'license-status error';
        return;
    }
    
    // 验证授权码格式（Gumroad格式：XXXXXX-XXXXXXXX-XXXXXXXX-XXXXXXXX）
    const licensePattern = /^[A-Z0-9]{6}-[A-Z0-9]{8}-[A-Z0-9]{8}-[A-Z0-9]{8}$/i;
    
    if (!licensePattern.test(license)) {
        status.textContent = '授权码格式不正确';
        status.className = 'license-status error';
        return;
    }
    
    // 可选：调用Gumroad API验证授权码（需要后端支持）
    // 这里简化为本地验证
    localStorage.setItem(LICENSE_STORAGE_KEY, license);
    status.textContent = '✓ 授权码激活成功！已升级到专业版';
    status.className = 'license-status success';
    input.value = '';
    updateUsageDisplay();
    
    // 3秒后刷新页面
    setTimeout(() => {
        location.reload();
    }, 2000);
}

// 在转换前检查使用限制
function checkBeforeConvert() {
    const usage = checkDailyUsage();
    
    if (!usage.allowed) {
        alert('您今天的免费使用次数已用完（3次）。\n\n升级到专业版即可无限使用！');
        // 滚动到付费区域
        document.querySelector('.pricing-container')?.scrollIntoView({ behavior: 'smooth' });
        return false;
    }
    
    return true;
}

// 页面加载时更新显示
document.addEventListener('DOMContentLoaded', function() {
    updateUsageDisplay();
});
```

### 🔧 常见问题和解决方案

#### Q1: 用户说授权码无效怎么办？
**解决方案**：
1. 确认用户输入的格式正确（包含连字符）
2. 在Gumroad后台查看该授权码状态
3. 如果授权码有效但无法激活，可能是浏览器缓存问题：
   - 让用户清空浏览器缓存
   - 或使用隐私模式/无痕模式

#### Q2: 如何处理退款？
**解决方案**：
1. 在Gumroad后台找到该订单
2. 点击 "Refund"
3. 选择退款金额（全额或部分）
4. 确认退款（自动处理）
5. Gumroad会自动撤销该授权码

#### Q3: 用户更换设备后授权码失效？
**解决方案**：
当前实现方案使用localStorage，更换设备需要重新输入授权码。
这是合理的，因为：
- 防止授权码被分享
- 每个购买对应一个授权码
- 用户保存好授权码邮件即可

#### Q4: 如何防止授权码被滥用？
**解决方案**：
1. 简单方案（当前）：依赖localStorage，每个浏览器需要输入
2. 进阶方案（需要后端）：
   - 调用Gumroad API验证授权码
   - 限制每个授权码的激活次数
   - 记录授权码使用的IP地址

#### Q5: 收入什么时候到账？
**解决方案**：
- Gumroad每周五自动转账到PayPal
- 需要销售额达到$10才会转账
- 如果不足$10，会累计到下周
